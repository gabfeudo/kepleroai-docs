<Steps>
    <Step title="Crea una nuova azione">
        1. Accedi alla sezione Addestramento nella dashboard di Keplero AI
        2. Clicca su **Integrazioni**
        3. Clicca su **+ Nuova integrazione**
        4. Seleziona **Hotel** dal menu laterale

        <Frame>
            <img src="/images/integrazioni/hotel/hotel-1.png" alt="Simple Booking" />
        </Frame>
    </Step>
    <Step title="Seleziona Simple Booking">
        Seleziona Simple Booking dall'elenco delle integrazioni disponibili.
    </Step>
    <Step title="Inserisci i dati necessari per collegarti a Simple Booking">
        <Frame>
            <img src="/images/integrazioni/hotel/simple-booking.png" alt="Simple Booking" />
        </Frame>
        Inserisci l'URL del motore.
    </Step>
</Steps>
